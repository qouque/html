뜨거운 바다에 마스크 쓰고 물놀이를 한다나..

		alt속성값이 커피인 img태그 서택
		img[alt="커피"]{
			border : 3px solid green;
		}
		#im1{
			border : 5px dotted hotpink;
		}
		
		id속성값이 im2인 img 선택
		img[id="im2"]{
			border : 10px inset green;
		}
		.cim{
			width : 150px;
			height : 150px;
			//id속성이 class 속성보다 우선순위가 높으므로
			//아래코드가 실행이 안된다.
			border : 10px dotted pink;
		}
